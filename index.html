import random print("⚽ Добро пожаловать в Эмодзи-Футбол!") team1 = "🔵 Синие" team2 = "🔴 Красные" score1 = 0 score2 = 0 for i in range(5): # 5 раундов (ударов) print(f"Раунд {i + 1}:") if random.choice([True, False]): print(f"{team1} забивают гол! 🥅⚽") score1 += 1 else: print(f"{team2} забивают гол! 🥅⚽") score2 += 1 # Итоговый счет print("\n🏁 Финальный свисток!") print(f"{team1}: {score1} | {team2}: {score2}") if score1 > score2: print(f"🎉 Победили {team1}!") elif score2 > score1: print(f"🎉 Победили {team2}!") else: print("🤝 Ничья!")
